// Load data from CSV file d3.csv("electric-vehicle-data.csv").then(function(data) { // Parse data data.forEach(function(d) { d.year = parseInt(d["Model Year"]); d.count = parseInt(d.Count); }); // Create chart var ctx = document.getElementById("myChart").getContext("2d"); var myChart = new Chart(ctx, { type: "line", data: { labels: data.map(function(d) { return d.year; }), datasets: [ { label: "Number of Electric Vehicles Released by Year", data: data.map(function(d) { return d.count; }), borderColor: "blue", fill: false } ] }, options: { responsive: true, maintainAspectRatio: false, scales: { yAxes: [ { ticks: { beginAtZero: true } } ] } } }); });
